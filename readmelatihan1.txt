#tipe data string
a="sutrisno"
b="cileungsi"
c=a+" "+b
print(c)

#tipe data number integer dan float
b=1
c=2.6
z=b+c
print(b)
print(z)
#konversi tipe data float ke int
print(int(z))

#tipe data tuple atau vektor
aa=(1,2,3,4,5)
aaa=(2,4)
print(aa)
print(aa+aaa)

#tipe data list
aa1=["sutrisno","lestari"]
aa2=["cileungsi","bogor"]
print(aa1+aa2)

#tipe data boolean
jk=True
if(jk):
    print("pria")
print(type(jk))

#tipe data boolean
jk1=False
if(jk1==1):
    print("pria")
else:
    print("wanita")
print(type(jk1))

#struktur data dictionari
zz={"sutrisno":"cileungsi","umur":25,"tinggi":182}
print(zz)
